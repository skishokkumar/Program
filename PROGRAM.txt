   program
  #include<stdio.h>
  int main()
  {
   int a;
   float b;
   double c;
   char d;
   scanf("%d",&a);
   scanf("%f",&b);
   scanf("%lf",&c);
   scanf("%c",&d);
   printf("%d",a);
   printf("%f",b);
   printf("%lf",c);
   printf("%c",d);
   return 0;
 }

  <---------------------------------------------------------------------datatypes----------------------------------------------------------------->
  what is this int,float,double,char?
     These are datatypes of the c program
     if we want to declare a number  we use int---------->4 byte 
     if we want to declare a decimal  we use float-------->4 byte
     if we want to declare a decimal  we use double----->8 byte
     if we want to declare a single character  we use char because the allocation of this char is only 1 byte
  <----------------------------------------------------------------------format specifiers-------------------------------------------------------->
   what is this %d,%f,%lf,%c?
         These are called format specifiers for int %d,float %f ,double %lf,char %c
